--SQL TO CREATE TEH EAV DATABASE
--RUN FIRST

CREATE TABLE Entity (
    ID INT PRIMARY KEY IDENTITY,
    Name VARCHAR(50) NOT NULL
);

CREATE TABLE Attribute (
    ID INT PRIMARY KEY IDENTITY,
    EntityID INT NOT NULL,
    Name VARCHAR(50) NOT NULL,
    Type VARCHAR(50) NOT NULL
);

CREATE TABLE Value (
    ID INT PRIMARY KEY IDENTITY,
    AttributeID INT NOT NULL,
    EntityID INT NOT NULL,
    Data VARCHAR(50) NOT NULL,
);
